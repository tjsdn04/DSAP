  • 집합 자료형 활용
# 집합 자료형 활용
s1 = set([1, 2, 3, 4, 5, 6])
s2 = set([4, 5, 6, 7, 8, 9])

# 교집합
print(s1 & s2)
print(s1.intersection(s2))

# 합집합
print(s1 | s2)
print(s1.union(s2))

# 차집합
print(s1 - s2)
print(s1.difference(s2))

# 중복 원소 확인
print(s1.isdisjoint(s2))

# 부분 집합 확인
print(s1.issubset(s2))
print(s1.issuperset(s2))
  • 수정
